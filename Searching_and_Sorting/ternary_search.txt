#include <bits/stdc++.h>
using namespace std;

bool ternary_search(vector<int> vec, int ele)
{
    int n = vec.size();
    int low = 0;
    int high = n - 1;
    
    while(low < high)
    {
        int mid1 = (low + high) *  1/3;
        int mid2 = (low + high) *  2/3;
        
        if(vec[mid1] == ele)
            return true;
        else
            if(vec[mid2] == ele)
                return true;
        else
            if(ele < vec[mid1])
                high = mid1 - 1;
        else
            if(ele > vec[mid1] && ele < vec[mid2])
            {
                low = mid1 + 1;
                high = mid2 - 1;
            }
        else
            if(ele > mid2)
                low = mid2 + 1;
    }
    
    if(vec[low] == ele)
        return true;
    else
        return false;
}

int main() {
    vector<int> vec = {1, 2, 4, 5};
	cout<<ternary_search(vec, 1);
	return 0;
}
