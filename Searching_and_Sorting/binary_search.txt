#include <bits/stdc++.h>
using namespace std;

bool binary_search(vector<int> vec, int ele)
{
    int n = vec.size();
    int low = 0;
    int high = n - 1;
    
    while(low < high)
    {
        int mid = (low + high)/2;
        
        if(vec[mid] == ele)
            return true;
        else
            if(ele < vec[mid])
                high = mid - 1;
        else
            if(ele > vec[mid])
                low = mid + 1;
    }
    
    return false;
}

int main() {
    vector<int> vec = { 1, 2, 4, 5};
	cout<<binary_search(vec, 2);
	return 0;
}
